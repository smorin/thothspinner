## Deliverable
```python
from thothspinner.textual import TextualThothSpinner
from textual.app import App, ComposeResult

class MyApp(App):
    def compose(self) -> ComposeResult:
        self.thoth = TextualThothSpinner(
            config={
                "theme": "dark",
                "spinner_style": "npm_dots",
                "colors": {
                    "primary": "#00FFFF",
                    "success": "#00FF00",
                    "error": "#FF0000"
                }
            }
        )
        yield self.thoth
    
    async def on_mount(self) -> None:
        # Access individual components
        self.thoth.spinner.start()
        self.thoth.timer.start()
        self.thoth.message.set("Processing data...")
        
        # Perform work
        for i in range(100):
            self.thoth.progress.set(i)
            await asyncio.sleep(0.1)
        
        # Global success state
        self.thoth.success("Operation complete!")
```
