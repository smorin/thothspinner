## Deliverable
```python
from thothspinner.textual.widgets import HintWidget
from textual.app import App, ComposeResult

class MyApp(App):
    def compose(self) -> ComposeResult:
        # Basic usage
        yield HintWidget(text="Press ESC to cancel", color="#FF5555")
        # With icon
        yield HintWidget(text="Loading", icon="⚠", color="#FFA500")
        # From config
        config = {"text": "Ready", "color": "#00FF00", "visible": True}
        yield HintWidget.from_config(config)
```
