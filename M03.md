## Deliverable

```python
from thothspinner.rich.components import ProgressComponent, TimerComponent
from rich.console import Console
from rich.live import Live
import time

console = Console()

# Progress component with percentage
progress = ProgressComponent(
    current=0,
    total=100,
    format={"style": "percentage"},
    color="#55FF55"
)

# Timer component with auto format
timer = TimerComponent(
    format={"style": "auto", "precision": 1},
    color="#FFFF55"
)

# Combined display
with Live(console.renderables([progress, timer]), console=console, refresh_per_second=10):
    timer.start()
    for i in range(101):
        progress.set(i)
        time.sleep(0.1)
    timer.stop()
    progress.success()
```
