# Milestone M02: Spinner Component (v0.2.0)

## Goal/Requirement
Implement the animated Spinner component as the second building block of ThothSpinner.
- Create frame-based animation system with configurable intervals
- Support multiple spinner styles (NPM dots, Claude stars, custom)
- Add state transformation capabilities (in_progress, success, error)
- Integrate with Rich's rendering system for smooth animation

## Out of Scope
- Textual component implementation
- Other components (Progress, Timer, Message)
- Parent orchestration/ThothSpinner container
- Shimmer effects (reserved for Message component)
- PyPI packaging and publishing

## Key Implementation Details

### SpinnerComponent Class (spinner.py)
```python
from rich.console import Console, ConsoleOptions, RenderResult
from rich.text import Text
from rich.style import Style
from typing import Optional, Dict, Any, List
import time

from thothspinner.core.states import ComponentState
from thothspinner.rich.spinners.frames import SPINNER_FRAMES

class SpinnerComponent:
    """Animated spinner component for Rich console."""
    
    def __init__(
        self,
        frames: Optional[List[str]] = None,
        interval: float = 0.08,
        color: str = "#D97706",
        style: str = "npm_dots",
        success_icon: str = "✓",
        error_icon: str = "✗",
        visible: bool = True,
        speed: float = 1.0,
        **kwargs
    ):
        self.frames = frames or SPINNER_FRAMES.get(style, SPINNER_FRAMES["npm_dots"])
        self.interval = interval
        self.color = color
        self.success_icon = success_icon
        self.error_icon = error_icon
        self.visible = visible
        self.speed = speed
        self._state = ComponentState.IN_PROGRESS
        self._start_time: Optional[float] = None
    
    @property
    def state(self) -> ComponentState:
        return self._state
    
    def start(self) -> None:
        """Start spinner animation."""
        self._state = ComponentState.IN_PROGRESS
        self._start_time = None  # Reset for fresh animation
    
    def success(self, message: Optional[str] = None) -> None:
        """Transition to success state."""
        self._state = ComponentState.SUCCESS
    
    def error(self, message: Optional[str] = None) -> None:
        """Transition to error state."""
        self._state = ComponentState.ERROR
    
    def reset(self) -> None:
        """Reset to in_progress state."""
        self._state = ComponentState.IN_PROGRESS
        self._start_time = None
    
    def __rich_console__(
        self, console: Console, options: ConsoleOptions
    ) -> RenderResult:
        if not self.visible:
            return
        
        # Determine what to display based on state
        if self._state == ComponentState.SUCCESS:
            style = Style(color="#00FF00")
            yield Text(self.success_icon, style=style)
        elif self._state == ComponentState.ERROR:
            style = Style(color="#FF0000")
            yield Text(self.error_icon, style=style)
        else:
            # Calculate frame based on time (stateless approach)
            current_time = console.get_time() if hasattr(console, 'get_time') else time.time()
            
            if self._start_time is None:
                self._start_time = current_time
            
            elapsed = (current_time - self._start_time) * self.speed
            frame_no = int(elapsed / self.interval) % len(self.frames)
            
            style = Style(color=self.color)
            yield Text(self.frames[frame_no], style=style)
```

def validate_frames(frames: list[str]) -> bool:
    """Validate that frame list is non-empty and contains strings."""
    return (
        isinstance(frames, list) and 
        len(frames) > 0 and 
        all(isinstance(f, str) and len(f) > 0 for f in frames)
    )

### Component States (states.py)
```python
"""Core state definitions for ThothSpinner components."""

from enum import Enum, auto

class ComponentState(Enum):
    """Component lifecycle states."""
    IN_PROGRESS = auto()  # Active/animating state
    SUCCESS = auto()      # Successful completion
    ERROR = auto()        # Error/failure state
    
    def can_transition_to(self, new_state: 'ComponentState') -> bool:
        """Check if transition to new_state is valid."""
        # Can always reset to IN_PROGRESS
        if new_state == ComponentState.IN_PROGRESS:
            return True
        
        # Can only go to terminal states from IN_PROGRESS
        if self == ComponentState.IN_PROGRESS:
            return new_state in (ComponentState.SUCCESS, ComponentState.ERROR)
        
        # Cannot transition between SUCCESS and ERROR directly
        return False
```

### Example Demo (spinner_demo.py)
```python
"""Demonstration of SpinnerComponent capabilities."""

from rich.console import Console
from rich.live import Live
from rich.table import Table
from rich.text import Text
from thothspinner.rich.components import SpinnerComponent
import time
import signal
import sys

def main():
    console = Console()
    
    # Demo 1: Different spinner styles
    console.print("[bold cyan]Spinner Styles Demo[/bold cyan]\n")
    
    styles = ["npm_dots", "claude_stars", "classic", "dots", "arrows"]
    for style in styles:
        spinner = SpinnerComponent(style=style)
        # Create a table to show label and spinner side by side
        table = Table.grid(padding=1)
        table.add_row(Text(f"{style}:"), spinner)
        
        with Live(table, console=console, refresh_per_second=20, transient=True):
            time.sleep(2)
    
    # Demo 2: State transitions (proper Live usage)
    console.print("\n[bold cyan]State Transitions Demo[/bold cyan]\n")
    
    spinner = SpinnerComponent()
    status_text = Text("Processing...")
    
    # Combine spinner and text in a table
    display = Table.grid(padding=1)
    display.add_row(spinner, status_text)
    
    with Live(display, console=console, refresh_per_second=20) as live:
        # Processing state
        time.sleep(2)
        
        # Success state
        spinner.success()
        status_text.plain = "Done!"
        time.sleep(1)
        
        # Reset and retry
        spinner.reset()
        status_text.plain = "Retrying..."
        time.sleep(2)
        
        # Error state
        spinner.error()
        status_text.plain = "Failed!"
        time.sleep(1)
    
    # Demo 3: Custom frames with interrupt handling
    console.print("\n[bold cyan]Custom Frames Demo (Press Ctrl+C to stop)[/bold cyan]\n")
    
    custom_frames = ["🌍", "🌎", "🌏"]
    spinner = SpinnerComponent(frames=custom_frames, interval=0.3)
    
    def signal_handler(signum, frame):
        """Handle interrupt gracefully."""
        spinner.error()
        console.print("\n[yellow]Interrupted by user[/yellow]")
        sys.exit(0)
    
    signal.signal(signal.SIGINT, signal_handler)
    
    with Live(spinner, console=console, refresh_per_second=10, transient=True):
        try:
            time.sleep(10)
            spinner.success()
            time.sleep(0.5)
        except KeyboardInterrupt:
            pass  # Handled by signal handler

if __name__ == "__main__":
    main()
```
