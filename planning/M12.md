## Deliverable
```python
from thothspinner.textual import TextualThothSpinner
from textual.app import App, ComposeResult

class MyApp(App):
    def compose(self) -> ComposeResult:
        self.thoth = TextualThothSpinner(
            spinner_style="npm_dots",
            hint_text="(esc to cancel)",
            success_duration=2.0,
        )
        yield self.thoth

    async def on_mount(self) -> None:
        # Start spinner and timer
        self.thoth.start()

        # Access individual components
        self.thoth.set_message(text="Processing data...")

        # Update progress
        for i in range(100):
            self.thoth.update_progress(current=i, total=100)
            await asyncio.sleep(0.1)

        # Global success state (propagates to all children)
        self.thoth.success("Operation complete!")
```
