```python
# src/thothspinner/core/states.py
from enum import Enum, auto
from typing import Optional

class ComponentState(Enum):
    """Component lifecycle states with validation."""
    IN_PROGRESS = auto()
    SUCCESS = auto()
    ERROR = auto()

    def can_transition_to(self, new_state: 'ComponentState') -> bool:
        """Validate state transitions per PRD rules."""
        if new_state == ComponentState.IN_PROGRESS:
            return True  # Can always reset
        if self == ComponentState.IN_PROGRESS:
            return new_state in (ComponentState.SUCCESS, ComponentState.ERROR)
        return False  # No direct SUCCESS <-> ERROR
```

**Rationale**: Enum provides type safety and prevents invalid state strings. The transition validation ensures state machine integrity.

#### Task M02-T04: Implement __rich_console__ Rendering with Frame Cycling
**File**: src/thothspinner/rich/components/spinner.py:101-200

**Implementation Details**:
```python
def _calculate_frame(self, current_time: float) -> int:
    """Calculate current frame index based on time."""
    if self._start_time is None:
        self._start_time = current_time
        return 0

    # Rich-style frame calculation with speed multiplier
    elapsed = (current_time - self._start_time) * self.speed
    frame_no = int(elapsed / self.interval) % len(self.frames)

    return frame_no
```

**Note**: Removed `_frame_offset` as it adds unnecessary state. Pure time-based calculation is simpler and more reliable.

```python
@classmethod
def validate_frames(cls, frames: List[str]) -> bool:
    """Validate frames are single-width characters for proper alignment."""
    from wcwidth import wcwidth
    for frame in frames:
        if wcwidth(frame) != 1:
            return False
    return True

def __rich_console__(
    self, console: Console, options: ConsoleOptions
) -> RenderResult:
    """Render spinner for Rich console."""
    if not self.visible:
        return

    # Get current time from console for consistency
    current_time = console.get_time() if hasattr(console, 'get_time') else time.time()

    if self._state == ComponentState.SUCCESS:
        style = Style(color="#00FF00")
        yield Text(self.success_icon, style=style)
    elif self._state == ComponentState.ERROR:
        style = Style(color="#FF0000")
        yield Text(self.error_icon, style=style)
    else:
        # Animate spinner using time-based calculation
        frame_idx = self._calculate_frame(current_time)
        style = Style(color=self.color)
        yield Text(self.frames[frame_idx], style=style)
```

**Rich Best Practice**: Validate single-width characters to ensure proper terminal alignment. Use console.get_time() for synchronized animations.

### Phase 3: State Management

```python
@property
def state(self) -> ComponentState:
    """Current component state."""
    return self._state

def start(self) -> None:
    """Start or restart spinner animation."""
    self._state = ComponentState.IN_PROGRESS
    self._start_time = None  # Reset animation

def success(self, message: Optional[str] = None) -> None:
    """Transition to success state."""
    if not self._state.can_transition_to(ComponentState.SUCCESS):
        # Log warning but don't raise exception
        return
    self._state = ComponentState.SUCCESS

def error(self, message: Optional[str] = None) -> None:
    """Transition to error state."""
    if not self._state.can_transition_to(ComponentState.ERROR):
        return
    self._state = ComponentState.ERROR
```

**Rationale**: Graceful handling of invalid transitions prevents runtime errors in production while maintaining state integrity.

```python
from rich.table import Table
from rich.text import Text

def styles_demo(console: Console):

def state_transitions_demo(console: Console):
    """Demo state transitions with proper Live usage."""
    console.print("\n[bold cyan]State Transitions Demo[/bold cyan]\n")

    spinner = SpinnerComponent()
    status_text = Text("Processing...")

    # Combine spinner and text in a table
    display = Table.grid(padding=1)
    display.add_row(spinner, status_text)

    with Live(display, console=console, refresh_per_second=20) as live:
        time.sleep(2)

        spinner.success()
        status_text.plain = "Done!"
        time.sleep(1)

        spinner.reset()
        status_text.plain = "Retrying..."
        time.sleep(2)

        spinner.error()
        status_text.plain = "Failed!"
        time.sleep(1)

def interrupt_demo(console: Console):
    """Demo interrupt handling in user code."""
    console.print("\n[bold cyan]Interrupt Demo (Press Ctrl+C)[/bold cyan]\n")

    spinner = SpinnerComponent(frames=["🌍", "🌎", "🌏"], interval=0.3)

        spinner.error()
        console.print("\n[yellow]Interrupted by user[/yellow]")
        sys.exit(0)

    with Live(spinner, console=console, refresh_per_second=10, transient=True):
        try:
            time.sleep(10)
            spinner.success()
        except KeyboardInterrupt:
            pass  # Handled by signal handler
```

## Anti-Patterns to Avoid

Based on Rich's architecture, avoid these common mistakes:

### 1. **Stateful Frame Tracking**
❌ **Don't do this:**
```python
def __rich_console__(self, console, options):
    self._current_frame = (self._current_frame + 1) % len(self.frames)
    return Text(self.frames[self._current_frame])
```

✅ **Do this instead:**
```python
def __rich_console__(self, console, options):
    current_time = console.get_time()
    frame_no = int((current_time - self._start_time) / self.interval) % len(self.frames)
    return Text(self.frames[frame_no])
```

### 2. **Component-Level Threading**
❌ **Don't manage threads in the component**
✅ **Let Live handle all threading and refresh cycles**

### 3. **Direct Console Printing in Live**
❌ **Don't mix console.print() with Live updates**
✅ **Use Table.grid() or Group to combine renderables**

### 4. **Complex Locking**
❌ **Don't use complex RLock patterns for simple state**
✅ **Keep state updates atomic and simple**

### 5. **High FPS Targets**
❌ **Don't target 60+ FPS for terminal animations**
✅ **Use 20 FPS as the standard for smooth terminal display**
