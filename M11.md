## Deliverable
```python
from thothspinner.textual.widgets import MessageWidget
from textual.app import App, ComposeResult

class MyApp(App):
    def compose(self) -> ComposeResult:
        self.message = MessageWidget(
            action_words=["Loading", "Processing", "Analyzing"],
            message="your data...",
            shimmer=True,
            shimmer_direction="left_to_right",
            color="#00FFFF"
        )
        yield self.message
    
    async def on_mount(self) -> None:
        # Message will cycle through action words with shimmer
        await asyncio.sleep(10)
        self.message.set_message("Complete!")
```
