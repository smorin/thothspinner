from typing import Any, Dict

from rich.console import Console, ConsoleOptions, RenderResult


class HintComponent:
    """Static hint text component for Rich console."""

    def __init__(
        self,
        text: str = "(esc to interrupt)",
        color: str = "#888888",
        visible: bool = True,
        **kwargs: Any

        self._visible = visible

**Update:** `src/thothspinner/rich/components/hint.py`

Then add the method to the HintComponent class:
```python
def __rich_console__(
    self, console: Console, options: ConsoleOptions
) -> RenderResult:
    """Render the hint component for Rich console.
    
    Error handling: If Style creation fails (unlikely with validated color),
    fallback to unstyled text to ensure component always renders.
    """
    if self._visible:
        try:
            style = Style(color=self._color)
            yield Text(self._text, style=style)
        except Exception:
            # Fallback to unstyled text if style creation fails
            # This could happen if Rich's internals change
            yield Text(self._text)
```

"""Demonstration of the HintComponent."""

import time
from rich.console import Console
from rich.live import Live

from thothspinner.rich.components import HintComponent


def main():
    """Run the hint component demonstration."""
    console = Console()
    
    # Basic usage
    console.print("\n[bold cyan]Basic HintComponent Usage[/bold cyan]\n")
    hint = HintComponent()
    console.print("Default hint:", hint)
    
    # Custom text and colors
    hints = [
        HintComponent("Press ENTER to continue", color="#00FF00"),
        HintComponent("Press Q to quit", color="#FF0000"),
    ]
    
    for hint in hints:
        console.print(hint)
        time.sleep(0.5)
    
    # Error handling demonstration: Invalid colors fallback gracefully
    console.print("\n[bold cyan]Error Handling Example[/bold cyan]")
    invalid_hint = HintComponent("Safe despite bad color", color="not-a-color")
    console.print("Invalid color fallback:", invalid_hint)
    
    # Visibility toggle
    hint = HintComponent("This hint will toggle", color="#FF00FF")
    with Live(hint, console=console, refresh_per_second=2) as live:
        for i in range(3):
            time.sleep(1)
            hint.visible = not hint.visible
            live.update(hint)
    
    console.print("\n[bold green]Demo completed![/bold green]\n")


if __name__ == "__main__":
    main()

## Complete HintComponent Implementation

**File:** `src/thothspinner/rich/components/hint.py`
```python
"""Hint component for Rich console output."""

from typing import Any, Dict, Optional

from rich.console import Console, ConsoleOptions, RenderResult
from rich.style import Style
from rich.text import Text


class HintComponent:
    """Static hint text component for Rich console.
    
    A simple component that displays helper text with customizable
    color and visibility. Typically used to show keyboard shortcuts
    or status information.
    
    Attributes:
        text: The hint text to display
        color: Hex color code for the text (e.g., "#888888")
        visible: Whether the component should be rendered
    
    Example:
        >>> from rich.console import Console
        >>> console = Console()
        >>> hint = HintComponent("Press ESC to cancel", color="#FF5555")
        >>> console.print(hint)
    """

    def __init__(
        self,
        text: str = "(esc to interrupt)",
        color: str = "#888888",
        visible: bool = True,
        **kwargs: Any
    ) -> None:
        """Initialize the HintComponent.
        
        Args:
            text: The hint text to display. Defaults to "(esc to interrupt)".
            color: Hex color code for the text. Defaults to "#888888" (gray).
            visible: Whether the component should be rendered. Defaults to True.
            **kwargs: Additional keyword arguments (reserved for future use).
        """
        self._text = text
        self._color = self._validate_hex_color(color)
        self._visible = visible
    
    def _validate_hex_color(self, color: str) -> str:
        """Validate hex color format and return color or default.
        
        Args:
            color: Hex color string to validate
            
        Returns:
            Valid hex color or default #888888 if invalid
        """
        import re
        if not re.match(r'^#[0-9A-Fa-f]{6}$', color):
            return "#888888"  # Default fallback
        return color

    @property
    def text(self) -> str:
        """Get the hint text."""
        return self._text

    @text.setter
    def text(self, value: str) -> None:
        """Set the hint text."""
        self._text = value

    @property
    def color(self) -> str:
        """Get the color hex code."""
        return self._color

    @color.setter
    def color(self, value: str) -> None:
        """Set the color hex code."""
        self._color = value

    @property
    def visible(self) -> bool:
        """Get the visibility state."""
        return self._visible

    @visible.setter
    def visible(self, value: bool) -> None:
        """Set the visibility state."""
        self._visible = value

    @classmethod
    def from_config(cls, config: Dict[str, Any]) -> "HintComponent":
        """Create a HintComponent from a configuration dictionary.
        
        Args:
            config: Configuration dictionary with keys:
                - text (str): The hint text
                - color (str): Hex color code
                - visible (bool): Visibility state
        
        Returns:
            A new HintComponent instance configured from the dictionary.
        
        Example:
            >>> config = {"text": "Loading...", "color": "#00FF00", "visible": True}
            >>> hint = HintComponent.from_config(config)
        """
        return cls(
            text=config.get("text", "(esc to interrupt)"),
            color=config.get("color", "#888888"),
            visible=config.get("visible", True)
        )

    def __rich_console__(
        self, console: Console, options: ConsoleOptions
    ) -> RenderResult:
        """Render the hint component for Rich console.
        
        This method is called by Rich when rendering the component.
        
        Args:
            console: The Rich console instance
            options: Console rendering options
        
        Yields:
            Rich Text object with styled hint text if visible
        """
        if self._visible:
            style = Style(color=self._color)
            yield Text(self._text, style=style)

    def __repr__(self) -> str:
        """Return string representation of the component."""
        return (
            f"HintComponent(text={self._text!r}, "
            f"color={self._color!r}, visible={self._visible})"
        )
```
